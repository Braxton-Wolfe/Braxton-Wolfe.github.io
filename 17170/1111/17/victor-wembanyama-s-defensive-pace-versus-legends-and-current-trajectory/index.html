# Function to get first 127 games stats
get_stats <- function(df, name, debut_year) {
  df <- df %>%
    filter(!str_detect(Season, "Yrs|Avg")) %>%
    mutate(G = as.numeric(G), DWS = as.numeric(DWS), 
           `BLK%` = as.numeric(`BLK%`), DBPM = as.numeric(DBPM),
           `DRB%` = as.numeric(`DRB%`), MP = as.numeric(MP)) %>%
    arrange(Season) %>%
    mutate(cum_games = cumsum(G))
  
  df_127 <- df %>% filter(cum_games <= 127 | lag(cum_games, default = 0) < 127)
  
  tibble(
    Player = name,
    Era = debut_year,
    DWS = round(sum(df_127$DWS, na.rm = TRUE), 1),
    `BLK%` = round(weighted.mean(df_127$`BLK%`, df_127$MP, na.rm = TRUE), 1),
    `DRB%` = round(weighted.mean(df_127$`DRB%`, df_127$MP, na.rm = TRUE), 1),
    DBPM = round(weighted.mean(df_127$DBPM, df_127$MP, na.rm = TRUE), 1)
  )
}

# Get all stats with debut years
all_stats <- bind_rows(
  get_stats(wemby, "Victor Wembanyama", "2023-24"),
  get_stats(duncan, "Tim Duncan", "1997-98"),
  get_stats(hakeem, "Hakeem Olajuwon", "1984-85"),
  get_stats(dwight, "Dwight Howard", "2004-05"),
  get_stats(ad, "Anthony Davis", "2012-13")
)

# Find max values
max_dws <- max(all_stats$DWS)
max_blk <- max(all_stats$`BLK%`)
max_drb <- max(all_stats$`DRB%`)
max_dbpm <- max(all_stats$DBPM)

# Add categories led column
all_stats <- all_stats %>%
  mutate(
    `Categories Led` = (DWS == max_dws) + (`BLK%` == max_blk) + 
                       (`DRB%` == max_drb) + (DBPM == max_dbpm)
  ) %>%
  arrange(desc(`Categories Led`), desc(DWS))

print(all_stats)
